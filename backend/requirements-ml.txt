# Optional ML extras — the high-accuracy FinBERT tier of the news sentiment
# engine (backend/services/sentiment_engine.py).
#
# These are intentionally kept OUT of the core requirements.txt because `torch`
# pulls a multi-GB download (and platform-specific wheels). The sentiment engine
# lazily imports `transformers` and degrades gracefully to the TextBlob tier
# (then the lexical tier) when these are absent, so installing them is purely an
# accuracy upgrade.
#
# Install on top of the core deps:
#   pip install -r backend/requirements.txt -r backend/requirements-ml.txt
#
# First use downloads the `ProsusAI/finbert` weights from Hugging Face (~440 MB),
# cached under ~/.cache/huggingface afterwards.
transformers
torch

# Local fallback embedder for the private "second brain" RAG
# (backend/services/embeddings.py). The default embedding tier is the
# OpenAI-compatible /embeddings endpoint (Ollama's nomic-embed-text, etc.); this
# sentence-transformers model is used only when that endpoint has no embeddings
# route and BRAIN_EMBED_FALLBACK is on. First use downloads all-MiniLM-L6-v2
# (~90 MB). Depends on torch (above).
sentence-transformers
